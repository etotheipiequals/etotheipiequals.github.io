def YearFig(Year):
    fig, ax = plt.subplots(figsize=(4,4),dpi=200)

    str_year = Year
    df_year = df_color.loc[~df_color['code'].isnull() 
                                    & ~df_color['gdp'].isnull() 
                                    & ~df_color['life'].isnull() 
                                    & ~df_color['pop'].isnull() 
                                    & (df_color['year'] == str_year)]

    x = pd.Series(df_year['gdp']).values
    y = pd.Series(df_year['life']).values
    p = pd.Series(df_year['pop']).values
    c = pd.Series(df_year['color']).values
    t = pd.Series(df_year['country']).values

    x_data = x.astype("float")
    y_data = y.astype("float")
    num = np.size(x_data)

    ms = p.astype("float")/1e6

    for ii in range(0,num-1): # exclude last point which is the world
        ax.scatter(x_data[ii],y_data[ii],s = 2.5*ms[ii],alpha=0.75, facecolor = c[ii], edgecolors='none')
        if ms[ii] > 50:
            ax.text(x_data[ii],y_data[ii],t[ii],fontsize=4)

    ax.text(2e4,30,'Year = %.0f'%(Year),fontsize=24,color = cDUkk)
    plt.xlim([-7000,70000]) # for linear
    plt.ylim([25,87])
    plt.ylabel('Life expectancy (years)')
    plt.xlabel('GDP per capita (2011 US dollars)')
    return fig, "Year = %.0f " % (Year)

def YearData(Year):
    str_year = Year
    df_year = df_color.loc[~df_color['code'].isnull() 
                                    & ~df_color['gdp'].isnull() 
                                    & ~df_color['life'].isnull() 
                                    & ~df_color['pop'].isnull() 
                                    & (df_color['year'] == str_year)]

    x = pd.Series(df_year['gdp']).values
    y = pd.Series(df_year['life']).values
    p = pd.Series(df_year['pop']).values
    c = pd.Series(df_year['color']).values
    t = pd.Series(df_year['country']).values

    x_data = x.astype("float")
    y_data = y.astype("float")
    num = np.size(x_data)

    ms = p.astype("float")/1e6

    return x_data, y_data, ms, c, t

def update(frame_number):
    ax.clear()
    ax.set_xscale('log')

    Year = 1950 + frame_number
    x, y, ms, c, t = YearData(Year)
    scatter = ax.scatter(x, y, s = 1.5*ms, alpha=0.5, facecolor = c, edgecolors='none')
    num = np.size(ms)
    for ii in range(0,num):
        if ms[ii] > 50:
            ax.text(x[ii],y[ii],t[ii],fontsize=4)
    ax.text(3e3,30,'Year = %.0f'%(Year),fontsize=24,color = cDUkk)

    plt.xlim([400,100000])

    plt.ylabel('Life expectancy (years)')
