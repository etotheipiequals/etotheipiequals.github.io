from scipy.optimize import leastsq, curve_fit

def fit1(x,A,B):
    return A+B*x

A,B = curve_fit(fit1,ipcc_emissions,ipcc_temp)[0] # exclude last two years take only 2250 points

xfit = np.arange(0,ipcc_emissions[num-1],ipcc_emissions[num-1]/num)
yfit = A + B * xfit

Use this linear fit model different emissions scenarios. Plot both emission per year and temperature vs. total CO2. To beautify we shall use Ed Hawkins warning colors. https://matplotlib.org/matplotblog/posts/warming-stripes/

from matplotlib.colors import ListedColormap

fig, [ax2, ax1] = plt.subplots(1,2, figsize=(12, 6))
plt.rcParams.update({'font.size': 18})

cmap = ListedColormap([
    '#08306b', '#08519c', '#2171b5', '#4292c6',
    '#6baed6', '#9ecae1', '#c6dbef', '#deebf7',
    '#fee0d2', '#fcbba1', '#fc9272', '#fb6a4a',
    '#ef3b2c', '#cb181d', '#a50f15', '#67000d',
])

ax1.set_xlabel('Total CO$_2$ (Gt)')
ax1.set_ylabel('Heating ($^\circ$C)')
ax1.set_xlim(0,5700)
ax1.set_ylim(-0.3,3.0)

ax2.set_ylabel('Emissions (GtCO$_2$/year)')
